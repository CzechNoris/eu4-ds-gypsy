	icon = IV_privileges_icon_2

	on_granted = {
		if = {
			limit = { 
				army_tradition = 100
			}
			create_conquistador = { tradition = 100 }
		}
		if = {
			limit = { 
				army_tradition = 90
				NOT = { army_tradition = 100 }
			}

		}
		if = {
			limit = { 
				army_tradition = 80

			}
			create_conquistador = { tradition = 80 }

			limit = { 
				army_tradition = 70

			}
			create_conquistador = { tradition = 70 }
		}
		if = {
			limit = { 
				army_tradition = 60
				NOT = { army_tradition = 70 }
			}
			create_conquistador = { tradition = 60 }

			limit = { 
				army_tradition = 50

			}
			create_conquistador = { tradition = 50 }
		}
		if = {
			limit = {
